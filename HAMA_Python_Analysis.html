query1 = """
SELECT 
    d.[Year],
    d.[Quarter],
    SUM(f.[FinalPrice]) AS TotalRevenue
FROM [dbo].[FactTransaction] AS f
JOIN [dbo].[DimDate] AS d
    ON f.[OrderDate] = d.[Date_SK]
GROUP BY d.[Year], d.[Quarter]
ORDER BY d.[Year], d.[Quarter];
"""

df_quarter = pd.read_sql(query1, conn)

top3 = df_quarter.nlargest(3, 'TotalRevenue')

x_labels = df_quarter['Quarter'].astype(str) + " " + df_quarter['Year'].astype(str)
plt.figure(figsize=(10,5))
bars = plt.bar(x_labels, df_quarter['TotalRevenue'], color='skyblue')

for i, bar in enumerate(bars):
    height = bar.get_height()
    label = df_quarter.loc[i, 'TotalRevenue']
    if df_quarter.loc[i, 'TotalRevenue'] in top3['TotalRevenue'].values:
        plt.text(
            bar.get_x() + bar.get_width() / 2,
            height + (height * 0.02),
            f"${label:,.0f}",
            ha='center', va='bottom', fontsize=9, fontweight='bold', color='darkblue'
        )

plt.title("Quarterly Membership Revenue")
plt.xlabel("Quarter")
plt.ylabel("Revenue ($)")
plt.xticks(rotation=45)
plt.tight_layout()
plt.show()

query_kpi1 = """
SELECT 
    d.[Year],
    d.[Month],
    SUM(f.[FinalPrice]) AS MonthlyRevenue
FROM [dbo].[FactTransaction] AS f
JOIN [dbo].[DimDate] AS d
    ON f.[OrderDate] = d.[Date_SK]
GROUP BY d.[Year], d.[Month]
ORDER BY d.[Year], d.[Month];
"""

# Pull data into DataFrame
df_monthly_revenue = pd.read_sql(query_kpi1, conn)
df_monthly_revenue['MonthLabel'] = (
    df_monthly_revenue['Month'].astype(str) + "-" + df_monthly_revenue['Year'].astype(str)
)

# Identify top 3 and bottom 3 months
top3 = df_monthly_revenue.nlargest(3, 'MonthlyRevenue')
bottom3 = df_monthly_revenue.nsmallest(3, 'MonthlyRevenue')

# Plot the line chart
plt.figure(figsize=(10,5))
plt.plot(df_monthly_revenue['MonthLabel'], df_monthly_revenue['MonthlyRevenue'], 
         marker='o', color='royalblue', linewidth=2)

plt.title("Monthly Revenue Trend")
plt.xlabel("Month-Year")
plt.ylabel("Revenue ($)")
plt.xticks(rotation=45)

# Add data labels for top 3 and bottom 3
for i, val in enumerate(df_monthly_revenue['MonthlyRevenue']):
    label = df_monthly_revenue.loc[i, 'MonthLabel']
    
    # Top 3: label above the line
    if val in top3['MonthlyRevenue'].values:
        plt.text(i, val + (val * 0.02), f"${val:,.0f}", 
                 ha='center', va='bottom', fontsize=9, 
                 color='darkgreen', fontweight='bold')
    
    # Bottom 3: label below the line
    elif val in bottom3['MonthlyRevenue'].values:
        plt.text(i, val - (val * 0.03), f"${val:,.0f}", 
                 ha='center', va='top', fontsize=9, 
                 color='crimson', fontweight='bold')

plt.tight_layout()
plt.show()
